   169	<script setup>
   170	import { api } from 'src/boot/axios'
   171	import { computed, ref, watch } from 'vue'
   172	
   173	const props = defineProps({
   174	  pasien: {
   175	    type: Object,
   176	    default: null
   177	  }
   178	})
   179	
   180	const loading = ref(false)
   181	const skriningData = ref([])
   182	const anamnesisData = ref([])
   183	const pemeriksaanFisikData = ref([])
   184	const refPrint = ref()
   185	
   186	const displayData = computed(() => {
   187	  const propData = props.pasien?.anamnesis_skrining
   188	  if (Array.isArray(propData) && propData.length) {
   189	    return propData
   190	  }
   191	  return skriningData.value
   192	})
   193	
   194	const latestAnamnesis = computed(() => {
   195	  const propData = props.pasien?.anamnesis
   196	  const source = Array.isArray(propData) && propData.length ? propData : anamnesisData.value
   197	  return Array.isArray(source) && source.length ? source[0] : null
   198	})
   199	
   200	const latestPemeriksaan = computed(() => {
   201	  const propData = props.pasien?.pemeriksaanfisik
   202	  const source = Array.isArray(propData) && propData.length ? propData : pemeriksaanFisikData.value
   203	  return Array.isArray(source) && source.length ? source[0] : null
   204	})
   205	
   206	function displayValue (value) {
   207	  if (value === null || value === undefined || value === '') return '-'
   208	  return value
   209	}
   210	
   211	function yesNoLabel (value) {
   212	  if (value === 1 || value === '1' || value === 'Iya' || value === 'iya' || value === 'Ya' || value === 'ya') {
   213	    return 'Ya'
   214	  }
   215	  if (value === 0 || value === '0' || value === 'Tidak' || value === 'tidak') {
   216	    return 'Tidak'
   217	  }
   218	  return '-'
   219	}
   220	
   221	function iconNyeri (anu) {
   222	  const val = typeof anu === 'string' ? (isNaN(parseInt(anu)) ? 0 : parseInt(anu)) : 0
   223	  if (val < 2) return 'icon-my-emoticon-excited-outline'
   224	  if (val >= 2 && val < 4) return 'icon-my-emoticon-outline'
   225	  if (val >= 4 && val < 6) return 'icon-my-emoticon-neutral-outline'
   226	  if (val >= 6 && val < 8) return 'icon-my-emoticon-confused-outline'
   227	  if (val >= 8 && val < 10) return 'icon-my-emoticon-angry-outline'
   228	  if (val === 10) return 'icon-my-emoticon-cry-outline'
   229	  return 'icon-my-emoticon-excited-outline'
   230	}
   231	
   232	function nyeriKeterangan (anu) {
   233	  const val = typeof anu === 'string' ? (isNaN(parseInt(anu)) ? 0 : parseInt(anu)) : 0
   234	  if (val === 0) return 'Tidak ada nyeri'
   235	  if (val > 0 && val <= 3) return 'Nyeri ringan'
   236	  if (val > 3 && val <= 6) return 'Nyeri sedang'
   237	  if (val > 6 && val <= 10) return 'Nyeri berat'
   238	  return '-'
   239	}
   240	
   241	function hasilJatuh (item) {
   242	  const seimbang = parseInt(item?.seimbang)
   243	  const penopang = parseInt(item?.penopang)
   244	  if (seimbang > 0 && penopang > 0) return 'Tinggi'
   245	  if (seimbang > 0 || penopang > 0) return 'Sedang'
   246	  return 'Rendah'
   247	}
   248	
   249	function detailJatuh (item) {
   250	  const seimbang = parseInt(item?.seimbang)
   251	  const penopang = parseInt(item?.penopang)
   252	  const seimbangText = seimbang > 0 ? 'tampak tidak seimbang (sempoyongan)' : 'tidak tampak seimbang (sempoyongan)'
   253	  const penopangText = penopang > 0 ? 'memegang pinggiran kursi atau benda lain sebagai penopang' : 'tidak memegang pinggiran kursi atau benda lain sebagai penopang'
   254	  return `Pasien ${seimbangText}. Pasien ${penopangText}.`
   255	}
   256	
   257	async function loadSkriningData () {
   258	  if (!props.pasien?.noreg) {
   259	    skriningData.value = []
   260	    return
   261	  }
   262	
   263	  const propData = props.pasien?.anamnesis_skrining
   264	  if (Array.isArray(propData) && propData.length) {
   265	    skriningData.value = propData
   266	    return
   267	  }
   268	
   269	  const propAnamnesis = Array.isArray(props.pasien?.anamnesis) ? props.pasien.anamnesis : []
   270	  const propPemeriksaan = Array.isArray(props.pasien?.pemeriksaanfisik) ? props.pasien.pemeriksaanfisik : []
   271	
   272	  if (propAnamnesis.length) anamnesisData.value = propAnamnesis
   273	  if (propPemeriksaan.length) pemeriksaanFisikData.value = propPemeriksaan
   274	
   275	  loading.value = true
   276	  try {
   277	    const response = await api.get('v1/simrs/pelayanan/skrining-rajal-kunjungan', {
   278	      params: { noreg: props.pasien.noreg }
   279	    })
   280	
   281	    const payload = response?.data || {}
   282	    const data = Array.isArray(payload?.data) ? payload.data : []
   283	    const anamnesis = Array.isArray(payload?.anamnesis) ? payload.anamnesis : []
   284	    const pemeriksaan = Array.isArray(payload?.pemeriksaanfisik) ? payload.pemeriksaanfisik : []
   285	
   286	    skriningData.value = data
   287	    anamnesisData.value = anamnesis
   288	    pemeriksaanFisikData.value = pemeriksaan
   289	  } catch (error) {
   290	    console.error('Gagal memuat data skrining rawat jalan', error)
   291	    skriningData.value = []
   292	    anamnesisData.value = []
   293	    pemeriksaanFisikData.value = []
   294	  } finally {
   295	    loading.value = false
   296	  }
   297	}
   298	
   299	watch(() => [
   300	  props.pasien?.noreg,
   301	  props.pasien?.anamnesis_skrining,
   302	  props.pasien?.anamnesis,
   303	  props.pasien?.pemeriksaanfisik
   304	], () => {
   305	  loadSkriningData()
   306	}, { immediate: true, deep: true })

   309	  id: 'pdfDoc',
   310	  previewTitle: 'SKRINING RAWAT JALAN',
   311	  popTitle: 'SKRINING RAWAT JALAN'
   312	}
